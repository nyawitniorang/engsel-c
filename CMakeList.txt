cmake_minimum_required(VERSION 3.10)
project(me-cli-sunset-c C)

set(CMAKE_C_STANDARD 11)

# Cari library
find_package(CURL REQUIRED)
find_package(PkgConfig REQUIRED)
pkg_check_modules(CJSON REQUIRED libcjson)

include_directories(${CURL_INCLUDE_DIRS} ${CJSON_INCLUDE_DIRS} src/include)

file(GLOB SOURCES "src/*.c" "src/app/*/*.c")

add_executable(me-cli-sunset ${SOURCES})
target_link_libraries(me-cli-sunset ${CURL_LIBRARIES} ${CJSON_LIBRARIES} m)
